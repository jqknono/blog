ШҜЩ„ЩҠЩ„ WireGuard Ш§Щ„ЩғШ§Щ…Щ„ Щ„Щ…ЩҲШ§Ш¬ЩҮШ© ЩӮЩҠЩҲШҜ QoS Щ„ЩҖ UDP Щ„ШҜЩү Щ…ШІЩҲШҜЩҠ Ш§Щ„Ш®ШҜЩ…Ш©

ЩҠЩҸШ№ШұЩҒ WireGuard ШЁШЁШіШ§Ш·ШӘЩҮ ЩҲЩғЩҒШ§ШЎШӘЩҮШҢ Щ„ЩғЩҶ Ш·ШұЩҠЩӮШ© Ш§Щ„Ш§ШӘШөШ§Щ„ Ш§Щ„ЩӮШ§ШҰЩ…Ш© Ш№Щ„Щү UDP ШӘШ¬Ш№Щ„ЩҮ Ш№ШұШ¶Ш© Щ„ЩӮЩҠЩҲШҜ QoS Щ…ЩҶ ЩӮШЁЩ„ Щ…ШІЩҲШҜЩҠ Ш§Щ„Ш®ШҜЩ…Ш©. ШіЩҠЩӮЩҲЩ… ЩҮШ°Ш§ Ш§Щ„Щ…ЩӮШ§Щ„ ШЁШӘШӯЩ„ЩҠЩ„ Ш№Щ…ЩҠЩӮ Щ„ШўЩ„ЩҠШ§ШӘ ШӘЩӮЩҠЩҠШҜ UDP Щ„ШҜЩү Щ…ШІЩҲШҜЩҠ Ш§Щ„Ш®ШҜЩ…Ш©ШҢ ЩҲЩҠЩӮШҜЩ… ШӯЩ„ЩҲЩ„Ш§ЩӢ Щ…ШӘШ№ШҜШҜШ© ШӘЩ… Ш§Щ„ШӘШӯЩӮЩӮ Щ…ЩҶЩҮШ§.

ШӘШӯЩ„ЩҠЩ„ ШўЩ„ЩҠШ§ШӘ QoS Щ„ЩҖ UDP Щ„ШҜЩү Щ…ШІЩҲШҜЩҠ Ш§Щ„Ш®ШҜЩ…Ш©

ЩҠШ·ШЁЩӮ Щ…ШІЩҲШҜЩҲ Ш§Щ„Ш®ШҜЩ…Ш© Ш№Ш§ШҜШ©ЩӢ ШіЩҠШ§ШіШ§ШӘ QoS ШЁЩҶШ§ШЎЩӢ Ш№Щ„Щү Ш®Щ…Ш§ШіЩҠ Ш§Щ„ШӘЩҠ (IP Ш§Щ„Щ…ШөШҜШұШҢ IP Ш§Щ„ЩҲШ¬ЩҮШ©ШҢ Ш§Щ„Щ…ЩҶЩҒШ° Ш§Щ„Щ…ШөШҜШұШҢ Ш§Щ„Щ…ЩҶЩҒШ° Ш§Щ„ЩҲШ¬ЩҮШ©ШҢ ЩҶЩҲШ№ Ш§Щ„ШЁШұЩҲШӘЩҲЩғЩҲЩ„):

  1. ЩғШҙЩҒ Ш§Щ„ШӯШІЩ… Ш§Щ„Ш№Щ…ЩҠЩӮ (DPI): Ш§Щ„ШӘШ№ШұЩҒ Ш№Щ„Щү Ш®ШөШ§ШҰШө ШӯШұЩғШ© Щ…ШұЩҲШұ VPN
  2. ШӘЩӮЩҠЩҠШҜ Ш§Щ„ШіШұШ№Ш© Ш№Щ„Щү Ш§Щ„Щ…ЩҶШ§ЩҒШ°: ШӘЩӮЩҠЩҠШҜ Ш№ШұШ¶ Ш§Щ„ЩҶШ·Ш§ЩӮ Ш§Щ„ШӘШұШҜШҜЩҠ Ш№Щ„Щү Ш§Щ„Щ…ЩҶШ§ЩҒШ° UDP ШәЩҠШұ Ш§Щ„ШҙШ§ШҰШ№Ш©
  3. ШӘЩӮЩҠЩҠШҜ Щ…ШҜШ© Ш§Щ„Ш§ШӘШөШ§Щ„: ЩҠШӘЩ… ШӘЩӮЩҠЩҠШҜ Ш§Щ„Ш§ШӘШөШ§Щ„Ш§ШӘ UDP Ш§Щ„Ш·ЩҲЩҠЩ„Ш© Ш§Щ„ШЈЩ…ШҜ
  4. ШӘШҙЩғЩҠЩ„ Ш§Щ„ШӯШұЩғШ©: ШӘШ№ШҜЩҠЩ„ ШЈЩҲЩ„ЩҲЩҠШ© ШӯШұЩғШ© Щ…ШұЩҲШұ ШЈЩҶЩҲШ§Ш№ ШЁШұЩҲШӘЩҲЩғЩҲЩ„Ш§ШӘ Щ…ШӯШҜШҜШ©

ШЁЩҠШ§ЩҶШ§ШӘ Ш§Щ„Ш§Ш®ШӘШЁШ§Шұ Ш§Щ„ЩҒШ№Щ„ЩҠ: ЩҒЩҠ ШҙШЁЩғШ© Ш§Щ„Ш§ШӘШөШ§Щ„Ш§ШӘШҢ ШЁШ№ШҜ ЩҶЩӮЩ„ ШӯШұЩғШ© UDP Щ…ШіШӘЩ…ШұШ© Щ„Щ…ШҜШ© 5 ШҜЩӮШ§ШҰЩӮШҢ ЩҠЩҶШ®ЩҒШ¶ Ш№ШұШ¶ Ш§Щ„ЩҶШ·Ш§ЩӮ Щ…ЩҶ 100 Щ…ЩҠШәШ§ШЁШӘ/Ш«Ш§ЩҶЩҠШ© ШҘЩ„Щү ШЈЩӮЩ„ Щ…ЩҶ 10 Щ…ЩҠШәШ§ШЁШӘ/Ш«Ш§ЩҶЩҠШ©

Щ…ЩӮШ§ШұЩҶШ© Ш§Щ„ШӯЩ„ЩҲЩ„

Ш§Щ„ШӯЩ„ШөШ№ЩҲШЁШ© Ш§Щ„ШӘЩҶЩҒЩҠШ°ЩҒЩӮШҜШ§ЩҶ Ш§Щ„ШЈШҜШ§ШЎЩӮШҜШұШ© Щ…ЩӮШ§ЩҲЩ…Ш© Ш§Щ„ШҘШәЩ„Ш§ЩӮШ§Щ„ШіЩҠЩҶШ§ШұЩҠЩҲ Ш§Щ„Щ…ЩҶШ§ШіШЁ
WireGuard over TCPвҳ…вҳ…вҳҶ20-30%вҳ…вҳ…вҳҶШЁЩҠШҰШ© ШҘШәЩ„Ш§ЩӮ ШөШ§ШұЩ…Ш©
Ш§Щ„Ш§ШіШӘЩ…Ш§Ш№ Ш№Щ„Щү Щ…ЩҶШ§ЩҒШ° Щ…ШӘШ№ШҜШҜШ©вҳ…вҳҶвҳҶ<5%вҳ…вҳ…вҳ…ШЁЩҠШҰШ© QoS Ш№Ш§ШҜЩҠШ©
ШӘШЁШҜЩҠЩ„ Ш§Щ„Щ…ЩҶШ§ЩҒШ° Ш§Щ„ШҜЩҠЩҶШ§Щ…ЩҠЩғЩҠвҳ…вҳ…вҳҶ<5%вҳ…вҳ…вҳ…вҳ…ШЁЩҠШҰШ© QoS Ш°ЩғЩҠШ©
Ш§ЩҶШӘШӯШ§Щ„ Ш§Щ„Щ…ЩҶШ§ЩҒШ° (ICMP/UDP)вҳ…вҳ…вҳ…10-15%вҳ…вҳ…вҳ…вҳ…ШЁЩҠШҰШ© ШҘШәЩ„Ш§ЩӮ Щ…ШӘЩӮШҜЩ…Ш©

Ш§Ш®ШӘШЁШ§Шұ Ш§Щ„ШЈШҜШ§ШЎ: Ш§Щ„ШӘШәЩ„ЩҠЩҒ ШЁЩҖ TCP ЩҠШӨШҜЩҠ ШҘЩ„Щү Ш§ЩҶШ®ЩҒШ§Ш¶ Ш§Щ„ШҘЩҶШӘШ§Ш¬ЩҠШ© ШЁЩҶШіШЁШ© 25% ШӘЩӮШұЩҠШЁШ§ЩӢШҢ ЩҲШІЩҠШ§ШҜШ© Ш§Щ„ШӘШЈШ®ЩҠШұ 15-20 Щ…Щ„Щ„ЩҠ Ш«Ш§ЩҶЩҠШ©

Ш§Щ„ШӯЩ„ Ш§Щ„Ш«Ш§ЩҶЩҠ: Ш§Щ„Ш§ШіШӘЩ…Ш§Ш№ Ш№Щ„Щү Щ…ЩҶШ§ЩҒШ° Щ…ШӘШ№ШҜШҜШ© + ШӘШЁШҜЩҠЩ„ ШҜЩҠЩҶШ§Щ…ЩҠЩғЩҠ (Ш§Щ„ШӯЩ„ Ш§Щ„Щ…ЩҲШөЩү ШЁЩҮ)

ШіЩғШұЩҠШЁШӘ ШӘШЁШҜЩҠЩ„ Ш§Щ„Щ…ЩҶШ§ЩҒШ° Ш§Щ„Ш°ЩғЩҠ Щ„Щ„Ш№Щ…ЩҠЩ„

# ШӯЩҒШё ЩғЩҖ wg-port-rotator.ps1
param(
    [int]$RangeStart = 51000,
    [int]$RangeEnd = 52000,
    [int]$ChangeInterval = 300  # ШӘШЁШҜЩҠЩ„ ЩғЩ„ 5 ШҜЩӮШ§ШҰЩӮ Ш§ЩҒШӘШұШ§Ш¶ЩҠШ§ЩӢ
)

# ЩғШҙЩҒ Щ…ШіШ§Шұ WireGuard ШӘЩ„ЩӮШ§ШҰЩҠШ§ЩӢ
$wgPath = if ($IsWindows) {
    "${env:ProgramFiles}\WireGuard\wg.exe"
} else {
    "/usr/bin/wg"
}

if (-not (Test-Path $wgPath)) {
    Write-Host "[ERROR] WireGuard ШәЩҠШұ Щ…Ш«ШЁШӘ ШЈЩҲ Щ…ШіШ§Шұ ШәЩҠШұ ШөШӯЩҠШӯ" -ForegroundColor Red
    exit 1
}

# Ш§Щ„ШӯШөЩҲЩ„ Ш№Щ„Щү Ш§Щ„ЩҲШ§Ш¬ЩҮШ© Ш§Щ„ЩҶШҙШ·Ш©
$interface = & $wgPath show interfaces
if (-not $interface) {
    Write-Host "[ERROR] Щ„Щ… ЩҠШӘЩ… Ш§Щ„Ш№Ш«ЩҲШұ Ш№Щ„Щү ЩҲШ§Ш¬ЩҮШ© WireGuard ЩҶШҙШ·Ш©" -ForegroundColor Red
    exit 1
}

# Ш§Щ„ШӯЩ„ЩӮШ© Ш§Щ„ШұШҰЩҠШіЩҠШ©
while ($true) {
    $peer = & $wgPath show $interface |
            Where-Object { $_ -match 'peer: ' } |
            Select-Object -First 1

    if (-not $peer) {
        Write-Host "[ERROR] Щ„Щ… ЩҠШӘЩ… Ш§Щ„Ш№Ш«ЩҲШұ Ш№Щ„Щү Ш№ЩӮШҜШ© Щ…ЩӮШ§ШЁЩ„Ш©" -ForegroundColor Red
        exit 1
    }

    $peerKey = $peer.Split()[1]
    $currentEndpoint = & $wgPath show $interface endpoints |
                      Where-Object { $_ -match $peerKey } |
                      ForEach-Object { $_.Split()[2] }

    $currentPort = if ($currentEndpoint) {
        [int]$currentEndpoint.Split(':')[-1]
    } else {
        $RangeStart
    }

    # ШӘЩҲЩ„ЩҠШҜ Щ…ЩҶЩҒШ° Ш№ШҙЩҲШ§ШҰЩҠ (Ш§ШіШӘШ«ЩҶШ§ШЎ Ш§Щ„Щ…ЩҶЩҒШ° Ш§Щ„ШӯШ§Щ„ЩҠ)
    $newPort = Get-Random -Minimum $RangeStart -Maximum ($RangeEnd + 1) -Exclude $currentPort

    # ШӘШӯШҜЩҠШ« Ш§Щ„ЩҶЩӮШ·Ш© Ш§Щ„ЩҶЩҮШ§ШҰЩҠШ©
    & $wgPath set $interface peer $peerKey endpoint "${currentEndpoint.Split(':')[0]}:$newPort"

    # Ш№ШұШ¶ ШӯШ§Щ„Ш© Ш§Щ„Ш§ШӘШөШ§Щ„
    & $wgPath show

    # Ш§Щ„Ш§ЩҶШӘШёШ§Шұ Щ„Щ„ШӘШЁШҜЩҠЩ„ Ш§Щ„ШӘШ§Щ„ЩҠ
    Start-Sleep -Seconds $ChangeInterval
}
