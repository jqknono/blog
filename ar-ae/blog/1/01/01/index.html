  name: whoami-deployment

  replicas: 3
  selector:
    matchLabels:
      app: whoami

    metadata:
      labels:
        app: whoami
    spec:
      containers:

  name: whoami-service

    app: whoami

  name: whoami-service

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: whoami-ingress
  namespace: default
spec:
  ingressClassName: external-lb-default
  rules:
    - host: whoami.example.com
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: whoami-service
                port:
                  number: 80
