# ШӘШӯЩ„ЩҠЩ„ Ш§ШіШӘШ¬Ш§ШЁШ© JSON
import json
import requests

def query_dns_json(domain="example.com", type="A"):
  """Ш§ШіШӘШ№Щ„Ш§Щ… DNS ШЁШ§ШіШӘШ®ШҜШ§Щ… ШӘЩҶШіЩҠЩӮ JSON"""
  url = "https://dns.google/resolve"
  params = {"name": domain, "type": type}
  headers = {"accept": "application/dns-json"}

  response = requests.get(url, params=params, headers=headers)
  return json.dumps(response.json(), indent=2)

# ШӘШӯЩ„ЩҠЩ„ Ш§ШіШӘШ¬Ш§ШЁШ© Wire Format
def query_dns_wire(domain="example.com"):
  """Ш§ШіШӘШ№Щ„Ш§Щ… DNS ШЁШ§ШіШӘШ®ШҜШ§Щ… ШӘЩҶШіЩҠЩӮ Wire Format"""
  import dns.message
  import requests
  import base64

  # ШҘЩҶШҙШ§ШЎ ШұШіШ§Щ„Ш© Ш§ШіШӘШ№Щ„Ш§Щ… DNS
  query = dns.message.make_query(domain, 'A')
  wire_format = query.to_wire()
  dns_query = base64.b64encode(wire_format).decode('utf-8')

  # ШҘШұШіШ§Щ„ Ш§Щ„Ш·Щ„ШЁ
  url = "https://dns.google/dns-query"
  params = {"dns": dns_query}
  headers = {"accept": "application/dns-message"}

  response = requests.get(url, params=params, headers=headers)
  dns_response = dns.message.from_wire(response.content)
  return str(dns_response)

if __name__ == "__main__":
  print("ЩҶШӘЩҠШ¬Ш© Ш§ШіШӘШ№Щ„Ш§Щ… JSON:")
  print(query_dns_json())
  print("\nЩҶШӘЩҠШ¬Ш© Ш§ШіШӘШ№Щ„Ш§Щ… Wire Format:")
  print(query_dns_wire())
