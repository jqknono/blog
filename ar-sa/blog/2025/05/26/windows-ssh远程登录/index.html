  1. еҗҜеҠЁ OpenSSH жңҚеҠЎпјҡ

    • е®үиЈ…е®ҢжҲҗеҗҺпјҢжү“ејҖе‘Ҫд»ӨжҸҗзӨәз¬ҰпјҲд»Ҙз®ЎзҗҶе‘ҳиә«д»ҪиҝҗиЎҢпјүгҖӮ
    • иҫ“е…Ҙ net start sshd жқҘеҗҜеҠЁ OpenSSH жңҚеҠЎгҖӮеҰӮжһңжғіиҰҒжҜҸж¬ЎејҖжңәж—¶иҮӘеҠЁеҗҜеҠЁиҜҘжңҚеҠЎпјҢеҸҜд»Ҙиҫ“е…Ҙ sc config sshd start= autoгҖӮ

  2. й…ҚзҪ®йҳІзҒ«еўҷпјҡ

    • зЎ®дҝқ Windows йҳІзҒ«еўҷе…Ғи®ё SSH иҝһжҺҘгҖӮеҸҜд»ҘйҖҡиҝҮвҖңжҺ§еҲ¶йқўжқҝвҖқ > вҖңзі»з»ҹе’Ңе®үе…ЁвҖқ > вҖңWindows Defender йҳІзҒ«еўҷвҖқ > вҖңй«ҳзә§и®ҫзҪ®вҖқпјҢз„¶еҗҺж–°е»әе…Ҙз«ҷи§„еҲҷпјҢе…Ғи®ё TCP з«ҜеҸЈ 22 зҡ„иҝһжҺҘгҖӮ

иҺ·еҸ– IP ең°еқҖе№¶иҝӣиЎҢиҝһжҺҘжөӢиҜ•

  1. иҺ·еҸ– IP ең°еқҖпјҡ

    • иҰҒд»ҺеҸҰдёҖеҸ°жңәеҷЁиҝһжҺҘеҲ°иҝҷеҸ°ејҖеҗҜдәҶ SSH жңҚеҠЎзҡ„ Windows з”өи„‘пјҢдҪ йңҖиҰҒзҹҘйҒ“е®ғзҡ„ IP ең°еқҖгҖӮеҸҜд»ҘеңЁе‘Ҫд»ӨжҸҗзӨәз¬ҰдёӢдҪҝз”Ё ipconfig е‘Ҫд»ӨжқҘжҹҘзңӢжң¬жңәзҡ„ IP ең°еқҖгҖӮ

  2. иҝһжҺҘжөӢиҜ•пјҡ

    • еңЁеҸҰдёҖеҸ°з”өи„‘жҲ–з§»еҠЁи®ҫеӨҮдёҠдҪҝз”Ё SSH е®ўжҲ·з«ҜпјҲдҫӢеҰӮпјҡPuTTYгҖҒTermius зӯүпјүе°қиҜ•иҝһжҺҘеҲ°дҪ зҡ„ Windows PCпјҢдҪҝз”Ёж јејҸ ssh username@your_ip_addressгҖӮе…¶дёӯ username жҳҜдҪ иҰҒзҷ»еҪ•зҡ„ Windows иҙҰжҲ·еҗҚпјҢyour_ip_address жҳҜдҪ д№ӢеүҚжҹҘеҲ°зҡ„ IP ең°еқҖгҖӮ

дҝ®ж”№й…ҚзҪ®

жіЁж„ҸйҒҝе…ҚдҪҝз”ЁеҜҶз Ғзҷ»еҪ•пјҢиҝҷжҳҜз»қеҜ№зҡ„йӣ·еҢәгҖӮеҠЎеҝ…дҪҝз”Ёе…¬й’ҘиҝӣиЎҢзҷ»еҪ•пјҢжҲ‘д»¬йңҖиҰҒдҝ®ж”№и®ҫзҪ®пјҢзҰҒз”ЁеҜҶз Ғзҷ»еҪ•пјҢе…Ғи®ёе…¬й’Ҙзҷ»еҪ•гҖӮ

иҜҘй…ҚзҪ®ж–Үд»¶дёҚдҫҝдҝ®ж”№пјҢйңҖиҰҒзү№ж®ҠжқғйҷҗжүҚиғҪдҝ®ж”№пјҢеҗҢж—¶иҝҳйңҖиҰҒдҝқиҜҒе…¶зӣ®еҪ•е’Ңж–Үд»¶зҡ„жқғйҷҗдёәзү№е®ҡеҖјпјҢиҝҷйҮҢжҺЁиҚҗдҪҝз”Ёи„ҡжң¬иҝӣиЎҢдҝ®ж”№гҖӮ

# жЈҖжҹҘз®ЎзҗҶе‘ҳжқғйҷҗ
$elevated = [bool]([System.Security.Principal.WindowsPrincipal]::new(
    [System.Security.Principal.WindowsIdentity]::GetCurrent()
).IsInRole([System.Security.Principal.WindowsBuiltInRole]::Administrator))

if (-not $elevated) {
    Write-Error "иҜ·д»Ҙз®ЎзҗҶе‘ҳиә«д»ҪиҝҗиЎҢжӯӨи„ҡжң¬"
    exit 1
}

# 1. жЈҖжҹҘе№¶е®үиЈ… OpenSSH жңҚеҠЎеҷЁ
Write-Host "жӯЈеңЁжЈҖжҹҘ OpenSSH жңҚеҠЎеҷЁе®үиЈ…зҠ¶жҖҒ..."
$capability = Get-WindowsCapability -Online -Name OpenSSH.Server~~~~0.0.1.0

if ($capability.State -ne 'Installed') {
    Write-Host "жӯЈеңЁе®үиЈ… OpenSSH жңҚеҠЎеҷЁ..."
    Add-WindowsCapability -Online -Name OpenSSH.Server~~~~0.0.1.0 | Out-Null
}

# 2. еҗҜеҠЁе№¶и®ҫзҪ®ејҖжңәиҮӘеҗҜ SSH жңҚеҠЎ
Write-Host "жӯЈеңЁй…ҚзҪ® SSH жңҚеҠЎ..."
$service = Get-Service sshd -ErrorAction SilentlyContinue
if (-not $service) {
    Write-Error "OpenSSH жңҚеҠЎе®үиЈ…еӨұиҙҘ"
    exit 1
}

if ($service.Status -ne 'Running') {
    Start-Service sshd
}
Set-Service sshd -StartupType Automatic

# 3. дҝ®ж”№й…ҚзҪ®ж–Үд»¶
$configPath = "C:\ProgramData\ssh\sshd_config"
if (Test-Path $configPath) {
    Write-Host "жӯЈеңЁеӨҮд»ҪеҺҹе§Ӣй…ҚзҪ®ж–Үд»¶..."
    Copy-Item $configPath "$configPath.bak" -Force
} else {
    Write-Error "жүҫдёҚеҲ°й…ҚзҪ®ж–Үд»¶: $configPath"
    exit 1
}

Write-Host "жӯЈеңЁдҝ®ж”№ SSH й…ҚзҪ®..."
$config = Get-Content -Path $configPath -Raw

# еҗҜз”Ёе…¬й’Ҙи®ӨиҜҒе№¶зҰҒз”ЁеҜҶз Ғзҷ»еҪ•
$config = $config -replace '^#?PubkeyAuthentication .*$','PubkeyAuthentication yes' `
                  -replace '^#?PasswordAuthentication .*$','PasswordAuthentication no'

# зЎ®дҝқеҢ…еҗ«еҝ…иҰҒй…ҚзҪ®
if ($config -notmatch 'PubkeyAuthentication') {
    $config += "`nPubkeyAuthentication yes"
}
if ($config -notmatch 'PasswordAuthentication') {
    $config += "`nPasswordAuthentication no"
}

# еҶҷеӣһй…ҚзҪ®ж–Үд»¶
$config | Set-Content -Path $configPath -Encoding UTF8

$userProfile = $env:USERPROFILE
$sshDir = Join-Path $userProfile ".ssh"
$authorizedKeysPath = Join-Path $sshDir "authorized_keys"
$PublicKeyPath = "D:\public_keys\id_rsa.pub"

# еҲӣе»ә .ssh зӣ®еҪ•
if (-not (Test-Path $sshDir)) {
    New-Item -ItemType Directory -Path $sshDir | Out-Null
}

# и®ҫзҪ® .ssh зӣ®еҪ•жқғйҷҗ
$currentUser = "$env:USERDOMAIN\$env:USERNAME"
$acl = Get-Acl $sshDir
$rule = New-Object System.Security.AccessControl.FileSystemAccessRule(
    $currentUser, "FullControl", "ContainerInherit,ObjectInherit", "None", "Allow"
)

Set-Acl $sshDir $acl

# ж·»еҠ е…¬й’Ҙ
if (Test-Path $PublicKeyPath) {
    $pubKey = Get-Content -Path $PublicKeyPath -Raw
    if ($pubKey) {
        # зЎ®дҝқе…¬й’Ҙжң«е°ҫжңүжҚўиЎҢз¬Ұ
        if (-not $pubKey.EndsWith("`n")) {
            $pubKey += "`n"
        }

        # иҝҪеҠ е…¬й’Ҙ
        Add-Content -Path $authorizedKeysPath -Value $pubKey -Encoding UTF8

        # и®ҫзҪ®ж–Үд»¶жқғйҷҗ
        $acl = Get-Acl $authorizedKeysPath
        $acl.SetSecurityDescriptorRule(
            (New-Object System.Security.AccessControl.FileSystemAccessRule(
                $currentUser, "FullControl", "None", "None", "Allow"
            ))
        )
        Set-Acl $authorizedKeysPath $acl
    }
} else {
    Write-Error "е…¬й’Ҙж–Үд»¶дёҚеӯҳеңЁ: $PublicKeyPath"
    exit 1
